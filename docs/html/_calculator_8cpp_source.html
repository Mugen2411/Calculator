110 ResultCode Calculator::_MakeNumberToken(TOKEN& token)
111 {
112 // トークン種別は数値
113 token.tag = TokenTag::Number;
114 // 仮数と指数
115 __int64 mantissa = 0;
116 __int64 exponent = 0;
117 // 整数かどうか
118 bool isInteger = true;
119 // 数値の抽出が終わったかどうか
120 bool isFinished = false;
121 do
122 {
123 switch (m_pExpression[m_currentIndex])
124 {
125 case '0':
126 case '1':
127 case '2':
128 case '3':
129 case '4':
130 case '5':
131 case '6':
132 case '7':
133 case '8':
134 case '9':
135 mantissa *= 10;
136 mantissa += m_pExpression[m_currentIndex] - '0';
137 if (!isInteger)
138 {
139 ++exponent;
140 }
141 break;
142 case '.':
143 // 既に小数点がいるのにまた小数点が来たら構文エラー
144 if (!isInteger)
145 {
146 return ResultCode::SyntaxError;
147 }
148 isInteger = false;
149 break;
150 default:
151 isFinished = true;
152 }
153 if (isFinished)
154 {
155 break;
156 }
157 } while (forwardIndex());
158
159 token.value.SetTop(mantissa);
160 __int64 bottom = 1;
161 for (int i = 0; i < exponent; ++i)
162 {
163 bottom *= 10;
164 }
165 token.value.SetBottom(bottom);
166 token.value.Reduce();
167 return ResultCode::Success;
168 }
169 ResultCode Calculator::_MakeOperatorToken(TOKEN& token)
170 {

175 else

177 const TokenTag lastTokenTag = m_tokenList.back().tag;

179 if (TokenTag::Number == lastTokenTag || TokenTag::BlockEnd == lastTokenTag)

193 token.operatorTag = OperatorTag::Plus;
194 break;
195 case '-':
196 token.operatorTag = OperatorTag::Minus;
197 break;
198 case '*':

205 break;

207 // 単項演算子になっていたらエラー

209 {
210 return ResultCode::SyntaxError;
211 }
212 token.operatorTag = OperatorTag::Divide;
213 break;

236 ResultCode Calculator::_ConvertToReversePolish()
237 {
238 std::stack<TOKEN> tokenStack;
239 TOKEN lastOp = {};
240 ResultCode resultCode = ResultCode::Success;
241 for (auto itr = m_tokenList.begin(); itr != m_tokenList.end();)
242 {
243 TOKEN cur = *itr;
244 itr = m_tokenList.erase(itr);
245 switch (cur.tag)
246 {
247 case TokenTag::Number:
248 m_reversePolish.emplace(cur);
249 resultCode = ResultCode::Success;
250 break;
251
252 case TokenTag::BlockBegin:
253 tokenStack.emplace(cur);
254 resultCode = ResultCode::Success;
255 break;
256
257 case TokenTag::UnaryOperator:
258 case TokenTag::BinaryOperator:
259 resultCode = _ConvertOperatorToReversePolish(cur, tokenStack);
260 break;
261
262 case TokenTag::BlockEnd:
263 resultCode = _ConvertBlockEndToReversePolish(cur, tokenStack);
264 break;
265
266 default:
267 return ResultCode::SyntaxError;
268 break;
269 }
270 if (ResultCode::Success != resultCode)
271 {
272 return resultCode;
273 }
274 }
275 while (!tokenStack.empty())
276 {
277 TOKEN lastOp = tokenStack.top();
278 if (TokenTag::BinaryOperator != lastOp.tag && TokenTag::UnaryOperator != lastOp.tag)
279 {
280 return ResultCode::SyntaxError;
281 }
282 m_reversePolish.emplace(lastOp);
283 tokenStack.pop();
284 }
285 return ResultCode::Success;

431 ResultCode Calculator::_CalclateBinaryOperator(const TOKEN operatorToken, const TOKEN& lhsToken, const TOKEN& rhsToken, TOKEN& ans)
432 {
433 if (lhsToken.tag != TokenTag::Number)
434 {
435 return ResultCode::SyntaxError;
436 }
437 if (rhsToken.tag != TokenTag::Number)
438 {
439 return ResultCode::SyntaxError;
440 }
441 ans.tag = TokenTag::Number;
442 ans.value = lhsToken.value;
443 switch (operatorToken.operatorTag)
444 {
445 case OperatorTag::Plus:
446 ans.value = ans.value + rhsToken.value;
447 break;
448 case OperatorTag::Minus:
449 ans.value = ans.value - rhsToken.value;
450 break;
451 case OperatorTag::Multiply:
452 ans.value = ans.value * rhsToken.value;
453 break;
454 case OperatorTag::Divide:
